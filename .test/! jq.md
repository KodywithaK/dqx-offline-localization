# jq -s -R
[
    split("\n")[]
    | sub(",";"???";"")
    | split("???")
]
| .[0] as $header
| .[1:]
| map(
    [$header, .]
    | transpose
    | map(
        {
            "key": (.[0]//""),
            "value": (.[1]//"")
            | sub("\\\\n";"\n";"g")
            #| sub("\\\"";"";"g")
            #| sub("\\\",";"")

    | from_entries
    | .value.["en"] = .SourceString
    | .value.["en"] = (.value.["en"] | sub("^\"";""))

    | .value.["en"] = (.value.["en"] | sub("\\\\";"";"g"))
    | .value.["en"] = (.value.["en"] | sub("''";"\"";"g"))

    | del(.SourceString)
)
| from_entries
